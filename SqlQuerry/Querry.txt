USE MovieDB

CREATE TABLE Titles(
    TConst INT PRIMARY KEY NOT NULL,
    TitleType VARCHAR(12) NOT NULL,
    PrimaryTitle VARCHAR(500) NOT NULL,
    OriginalTitle VARCHAR(500) NOT NULL,
    IsAdult BIT NOT NULL,
    StartYear SMALLINT NULL,
    EndYear SMALLINT NULL,
    RuntimeMinutes INT NULL
)

CREATE TABLE Genres(
    GenreID INT IDENTITY (1,1) PRIMARY KEY NOT NULL,
    Genre VARCHAR(20) NOT NULL UNIQUE
)

CREATE TABLE Titles_Genres(
    FK_TConst INT NOT NULL,
    FK_GenreID INT NOT NULL,
    PRIMARY KEY (FK_TConst, FK_GenreID),
    CONSTRAINT FK_TitlesGenres_TConst FOREIGN KEY (FK_TConst) REFERENCES Titles(TConst),
    CONSTRAINT FK_TitlesGenres_GenreID FOREIGN KEY (FK_GenreID) REFERENCES Genres(GenreID)
)

CREATE TABLE [Names](
    NConst INT PRIMARY KEY NOT NULL,
    PrimaryName  VARCHAR(110) NOT NULL,
    BirthYear SMALLINT NULL,
    DeathYear SMALLINT NULL
)

CREATE TABLE PrimaryProfessions(
    PPID INT PRIMARY KEY NOT NULL IDENTITY (1,1),
    PrimaryProfession VARCHAR(30) NOT NULL UNIQUE
)

CREATE TABLE Names_PrimaryProfessions(
	FK_NConst INT NOT NULL,
	FK_PPID INT NOT NULL,
	PRIMARY KEY (FK_NConst, FK_PPID),
	CONSTRAINT FK_NamesPrimaryProfessions_NConst FOREIGN KEY (FK_NConst) REFERENCES [Names](NConst),
	CONSTRAINT FK_NamesPrimaryProfessions_PPID FOREIGN KEY (FK_PPID) REFERENCES [PrimaryProfessions](PPID)
)


CREATE TABLE Names_Titles(
    FK_NConst INT NOT NULL,
    FK_TConst INT NOT NULL,
    PRIMARY KEY (FK_NConst, FK_TConst),
    CONSTRAINT FK_KnownForTitle_NConst FOREIGN KEY (FK_NConst) REFERENCES [Names](NConst),
    CONSTRAINT FK_KnownForTitle_TConst FOREIGN KEY (FK_TConst) REFERENCES Titles(TConst)
)

CREATE TABLE CrewRoles(
    RoleID INT PRIMARY KEY IDENTITY (1,1) NOT NULL,
    RoleName VARCHAR(30) NOT NULL UNIQUE 
)

CREATE TABLE Titles_Crew(
    FK_TConst INT NOT NULL,
    FK_NConst INT NOT NULL,
    FK_RoleID INT NOT NULL,
    PRIMARY KEY (FK_TConst, FK_NConst, FK_RoleID),
    CONSTRAINT FK_TitlesCrew_TConst FOREIGN KEY (FK_TConst) REFERENCES Titles(TConst),
    CONSTRAINT FK_TitlesCrew_NConst FOREIGN KEY (FK_NConst) REFERENCES [Names](NConst),
    CONSTRAINT FK_TitlesCrew_RoleID FOREIGN KEY (FK_RoleID) REFERENCES CrewRoles(RoleID)
)

CREATE OR ALTER PROCEDURE SearchMovieTitle 
    @TitleFragment VARCHAR(500)
AS
BEGIN
    SELECT 
        Titles.*, 
        Genres.Genre
    FROM 
        Titles
    JOIN 
        Titles_Genres ON Titles.TConst = Titles_Genres.FK_TConst
    JOIN 
        Genres ON Titles_Genres.FK_GenreID = Genres.GenreID
    WHERE 
        Titles.PrimaryTitle LIKE '%' + @TitleFragment + '%'
    ORDER BY 
        Titles.PrimaryTitle ASC;
END

CREATE OR ALTER PROCEDURE SearchName 
    @NameFragment VARCHAR(125)
AS
BEGIN
    SELECT 
        Names.*,
        PrimaryProfessions.PrimaryProfession,
        Titles.PrimaryTitle
    FROM 
        Names
    LEFT JOIN 
        Names_PrimaryProfessions ON Names.NConst = Names_PrimaryProfessions.FK_NConst
    LEFT JOIN 
        PrimaryProfessions ON Names_PrimaryProfessions.FK_PPID = PrimaryProfessions.PPID
   
    LEFT JOIN 
        Names_Titles ON Names.NConst = Names_Titles.FK_NConst
    LEFT JOIN 
        Titles ON Names_Titles.FK_TConst = Titles.TConst
        
    WHERE 
        Names.PrimaryName LIKE '%' + @NameFragment + '%'
    ORDER BY 
        Names.PrimaryName ASC;
END

CREATE OR ALTER PROCEDURE AddMovie 
    @TitleType VARCHAR(12), 
    @PrimaryTitle VARCHAR(500),
    @OriginalTitle VARCHAR(500), 
    @IsAdult BIT,
    @StartYear SMALLINT, 
    @EndYear SMALLINT,
    @RuntimeMinutes INT, 
    @Genres VARCHAR(75)
AS
BEGIN
    DECLARE @NewTConst INT;
    SELECT @NewTConst = MAX(TConst) + 1 FROM Titles;

    INSERT INTO Titles (TConst, TitleType, PrimaryTitle, OriginalTitle, IsAdult, StartYear, EndYear, RuntimeMinutes) 
    VALUES (@NewTConst, @TitleType, @PrimaryTitle, @OriginalTitle, @IsAdult, @StartYear, @EndYear, @RuntimeMinutes);

    -- Håndtér komma-separerede genrer
    IF @Genres IS NOT NULL
    BEGIN
        INSERT INTO Titles_Genres (FK_TConst, FK_GenreID)
        SELECT @NewTConst, g.GenreID
        FROM STRING_SPLIT(@Genres, ',') s
        JOIN Genres g ON g.Genre = TRIM(s.value);
    END
END

CREATE OR ALTER PROCEDURE AddPerson 
    @PrimaryName VARCHAR(110),
    @BirthYear SMALLINT = NULL, 
    @DeathYear SMALLINT = NULL,
    @Professions VARCHAR(500) = NULL,
    @KnownForTitles VARCHAR(500) = NULL
AS
BEGIN
    DECLARE @NewNConst INT;
    SELECT @NewNConst = MAX(NConst) + 1 FROM Names;

    INSERT INTO Names (NConst, PrimaryName, BirthYear, DeathYear) 
    VALUES (@NewNConst, @PrimaryName, @BirthYear, @DeathYear);

    IF @Professions IS NOT NULL
    BEGIN
        INSERT INTO Names_PrimaryProfessions (FK_NConst, FK_PPID)
        SELECT @NewNConst, pp.PPID

        FROM STRING_SPLIT(@Professions, ',') s
        JOIN PrimaryProfessions pp ON pp.PrimaryProfession = TRIM(s.value); 
    END

    IF @KnownForTitles IS NOT NULL
    BEGIN
        INSERT INTO Names_Titles (FK_NConst, FK_TConst)
        SELECT @NewNConst, t.TConst
        FROM STRING_SPLIT(@KnownForTitles, ',') s
        JOIN Titles t ON t.TConst = TRY_CAST(TRIM(s.value) AS INT);
    END
END

CREATE OR ALTER PROCEDURE UpdateMovie 
    @TConst INT,
    @TitleType VARCHAR(12), 
    @PrimaryTitle VARCHAR(500),
    @OriginalTitle VARCHAR(500), 
    @IsAdult BIT,
    @StartYear SMALLINT, 
    @EndYear SMALLINT,
    @RuntimeMinutes INT,
    @Genres VARCHAR(500)
AS
BEGIN
    UPDATE Titles
    SET 
        TitleType = @TitleType,
        PrimaryTitle = @PrimaryTitle,
        OriginalTitle = @OriginalTitle,
        IsAdult = @IsAdult,
        StartYear = @StartYear,
        EndYear = @EndYear,
        RuntimeMinutes = @RuntimeMinutes
    WHERE 
        TConst = @TConst;

    DELETE FROM Titles_Genres WHERE FK_TConst = @TConst;

    IF @Genres IS NOT NULL
    BEGIN
        INSERT INTO Titles_Genres (FK_TConst, FK_GenreID)
        SELECT @TConst, g.GenreID
        FROM STRING_SPLIT(@Genres, ',') s
        JOIN Genres g ON g.Genre = TRIM(s.value);
    END
END
